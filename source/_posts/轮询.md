    <script>
        var access = document.querySelector('#text').innerText;
        console.info('-----------', access);

        var count = 0;
        if (!Number(access)) {
            let timer = setInterval(_ => {
                count += 1;

                console.info('access', document.querySelector('#text').innerText);
                if (Number(document.querySelector('#text').innerText) || count === 20) {
                    
                    console.info('>>>>>>>>>>>>>>>>>>>>>', access, count);
                    clearInterval(timer);
                }
            }, 500);
        }


        document.getElementById('btn').addEventListener('click', function() {

            document.querySelector('#text').innerText = 1;
            console.info('access===========', document.querySelector('#text').innerText)
            
        })

        

    </script>
</body>
